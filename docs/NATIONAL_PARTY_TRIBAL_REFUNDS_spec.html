AGGREGATION GROUP A/N-100 X (error) NATIONAL_PARTY_HEADQUARTERS_ACCOUNT If Transaction Type is TRIBAL_REFUND_NP_HEADQUARTERS_ACCOUNT then Aggregation Group is NATIONAL_PARTY_HEADQUARTERS_ACCOUNT; If Transaction Type is TRIBAL_REFUND_NP_CONVENTION_ACCOUNT then Aggregation Group is NATIONAL_PARTY_CONVENTION_ACCOUNT; If Transaction Type is TRIBAL_REFUND_NP_RECOUNT then Aggregation Group is NATIONAL_PARTY_RECOUNT_ACCOUNT
  • AGGREGATION_GROUP = 'NATIONAL_PARTY_HEADQUARTERS_ACCOUNT' if TRANSACTION_TYPE_IDENTIFIER equals TRIBAL_REFUND_NP_HEADQUARTERS_ACCOUNT
  • AGGREGATION_GROUP = 'NATIONAL_PARTY_CONVENTION_ACCOUNT' if TRANSACTION_TYPE_IDENTIFIER equals TRIBAL_REFUND_NP_CONVENTION_ACCOUNT
  • AGGREGATION_GROUP = 'NATIONAL_PARTY_RECOUNT_ACCOUNT' if TRANSACTION_TYPE_IDENTIFIER equals TRIBAL_REFUND_NP_RECOUNT
  • REQUIRED
  • must be one of: ['NATIONAL_PARTY_HEADQUARTERS_ACCOUNT', 'NATIONAL_PARTY_CONVENTION_ACCOUNT', 'NATIONAL_PARTY_RECOUNT_ACCOUNT']

  • must be one of: ['Headquarters Buildings Account: Refund', 'Pres. Nominating Convention Account: Refund', 'Recount/Legal Proceedings Account: Refund']
